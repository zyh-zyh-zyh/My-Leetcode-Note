# 剑指Offer 36.二叉搜索树与双向链表

## My Solution

```java
/*
// Definition for a Node.
class Node {
    public int val;
    public Node left;
    public Node right;

    public Node() {}

    public Node(int _val) {
        val = _val;
    }

    public Node(int _val,Node _left,Node _right) {
        val = _val;
        left = _left;
        right = _right;
    }
};
*/
class Solution {
    Node less;
    Node max;
    Node min;
    public Node treeToDoublyList(Node root) {
        max = root;
        min = root;
        less = root;
        if (root == null) {
            return null;
        }
        Node tmp = root;
        while (tmp != null) {
            if (tmp.left == null) {
                min = tmp;
                break;
            }
            tmp = tmp.left;
        }
        tmp = root;
        while (tmp != null) {
            if (tmp.right == null) {
                max = tmp;
                break;
            }
            tmp = tmp.right;
        }
        less = min;
        dfs(root);
        min.left = max;
        max.right = min;
        return min;

    }
    
    public void dfs(Node cur) {
        if (cur == null) {
            return;
        }

        dfs(cur.left);
        if (cur != less) {
            cur.left = less;
            less.right = cur;
            less = cur;
        }
        dfs(cur.right);
    }

}

```
