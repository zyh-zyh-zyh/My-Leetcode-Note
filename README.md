### 617. 合并二叉树

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public TreeNode mergeTrees(TreeNode root1, TreeNode root2) {
        if (root1 == null && root2 == null) return root1;
        Queue<TreeNode> que1 = new LinkedList();
        Queue<TreeNode> que2 = new LinkedList();

        if (root1 != null && root2 != null) {
            root1.val += root2.val;
            que1.add(root1);
            que2.add(root2);
        }
        else if (root1 != null) {
            que1.add(root1);
            que2.add(new TreeNode(0));
        }
        else if (root2 != null) {
            TreeNode tmp = new TreeNode(0);
            root1 = tmp;
            root1.val += root2.val;
            que1.add(root1);
            que2.add(root2);
        }

        while (!que1.isEmpty() || !que2.isEmpty()) {
            TreeNode node1 = null;
            TreeNode node2 = null;
            if (!que1.isEmpty()) node1 = que1.remove();
            if (!que2.isEmpty()) node2 = que2.remove();

            if (node1.left != null && node2.left != null) {
                node1.left.val += node2.left.val;
                que1.add(node1.left);
                que2.add(node2.left);
            }
            else if (node1.left != null && node2.left == null) {
                que1.add(node1.left);
                que2.add(new TreeNode(0));
            }
            else if (node1.left == null && node2.left != null) {
                TreeNode tmp = new TreeNode(node2.left.val);
                node1.left = tmp;
                que1.add(tmp);
                que2.add(node2.left);
            }

            if (node1.right != null && node2.right != null) {
                node1.right.val += node2.right.val;
                que1.add(node1.right);
                que2.add(node2.right);
            }
            else if (node1.right != null && node2.right == null) {
                que1.add(node1.right);
                que2.add(new TreeNode(0));
            }
            else if (node1.right == null && node2.right != null) {
                TreeNode tmp = new TreeNode(node2.right.val);
                node1.right = tmp;
                que1.add(tmp);
                que2.add(node2.right);
            }
        }
        return root1;
    }
}
```
